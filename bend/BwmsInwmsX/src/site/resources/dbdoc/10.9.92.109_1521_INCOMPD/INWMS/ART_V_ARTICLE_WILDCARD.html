create or replace view art_v_article_wildcard as
(
select clt.id_client   as id_client,
       '*'             as id_article,
       null            as stat,
       null            as stat_erp,
       null            as txt_label,
       null            as txt_label_long1,
       null            as txt_label_long2,
       null            as code_ean,
       null            as unit_stock,
       null            as unit_wt,
       null            as unit_len,
       null            as unit_vol,
       null            as wt_gross,
       null            as wt_net,
       null            as len,
       null            as width,
       null            as height,
       null            as vol,
       null            as grp_articles,
       null            as cl_articles,
       null            as flg_batch,
       null            as flg_bbd,
       null            as bbd_calculation,
       null            as unit_bbd_calculation,
       null            as id_cre,
       null            as ts_cre,
       null            as id_upd,
       null            as ts_upd
  from wb_client clt
union
select art.id_client,
       art.id_article,
       art.stat,
       art.stat_erp,
       art.txt_label,
       art.txt_label_long1,
       art.txt_label_long2,
       art.code_ean,
       art.unit_stock,
       art.unit_wt,
       art.unit_len,
       art.unit_vol,
       art.wt_gross,
       art.wt_net,
       art.len,
       art.width,
       art.height,
       art.vol,
       art.grp_articles,
       art.cl_articles,
       art.flg_batch,
       art.flg_bbd,
       art.bbd_calculation,
       art.unit_bbd_calculation,
       art.id_cre,
       art.ts_cre,
       art.id_upd,
       art.ts_upd
  from art_article         art
);
