select soa.id_site           || '@' ||

       soa.id_order          || '@' ||
       soa.num_partial_order || '@' ||
       pa.id_package         || '@' as id_sas,

       soa.id_order,

       soa.num_item,
       soa.typ_shipment_external,
       soa.id_carrier_external,
       soa.ts_cre,

       sum(coalesce(pa.wt_weighted, pa.wt_gross,lou.wt_gross)) as Weight,

       tsa.storage_location,

       tsl.stat_occupied,
       tsl.flg_block_store_in,
       tsl.flg_block_store_out,
       tsl.pct_occupied,
       gi.stat,
       gi.typ_order,
       gi.priority_curr,
       gi.ts_shipping_target
from ship_order_asgmt soa
left join gi_order gi on gi.id_site = soa.id_site
            and gi.id_client = soa.id_client
            and gi.id_order = soa.id_order
            and gi.num_partial_order = soa.num_partial_order
left outer join load_loading loa on loa.id_site = soa.id_site
            and loa.id_loading = soa.id_loading
left outer join load_loading_staging_usage lsu on lsu.id_site  = soa.id_site
            and lsu.id_loading = soa.id_loading
            and lsu.num_item = soa.num_item
left outer join load_package_loading_asgmt lpa on lpa.id_site  = lsu.id_site
            and lpa.id_loading = lsu.id_loading
            and lpa.num_item = lsu.num_item
left outer join pa_package pa on pa.id_site  = lpa.id_site
            and pa.id_package = lpa.id_package
            and pa.typ_ref = 'ORDER'
            and pa.id_ref_1 = gi.id_site
            and pa.id_ref_2 = gi.id_client
            and pa.id_ref_3 = gi.id_order
            and pa.id_ref_4 = gi.num_partial_order
left outer join im_load_unit lou on lou.id_site = pa.id_site
            and lou.id_lu = pa.id_lu
left outer join top_staging_area tsa on tsa.id_site = lsu.id_site
            and tsa.id_staging_area = lsu.id_staging_area
left outer join top_storage_location tsl on tsl.id_site = tsa.id_site
            and tsl.storage_area = tsa.storage_area
            and tsl.storage_location = tsa.storage_location
group by soa.id_site,
