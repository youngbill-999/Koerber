create or replace view gr_v_setup_item as
select pr.ID_SITE,
       pr.ID_RECORD,
       pr.TYP,
       pr.PRIORITY,
       pr.TYP_STEP,
       pr.STAT_STEP,
       pr.ID_NODE,
       pr.TYP_REF,
       pr.ID_REF_1,
       pr.ID_REF_2,
       pr.ID_REF_3,
       pr.ID_REF_4,
       pr.ID_REF_5,
       pr.ID_REF_6,
       pr.ID_CLIENT,
       pr.ID_ARTICLE,
       pr.QTY,
       pr.ARTVAR,
       pr.TYP_STOCK,
       pr.TYP_LOCK,
       pr.ID_BATCH,
       pr.TS_BBD,
       pr.TYP_SPECIAL_STOCK,
       pr.ID_SPECIAL_STOCK,
       pr.STAT_QUALITY_CONTROL,
       pr.STAT_CUSTOMS,
       pr.SEP_CRIT_01,
       pr.SEP_CRIT_02,
       pr.SEP_CRIT_03,
       pr.SEP_CRIT_04,
       pr.SEP_CRIT_05,
       pr.SEP_CRIT_06,
       pr.SEP_CRIT_07,
       pr.SEP_CRIT_08,
       pr.SEP_CRIT_09,
       pr.SEP_CRIT_10,
       pr.WT,
       pr.UNIT_WT,
       pr.VOL,
       pr.UNIT_VOL,
       pr.ID_LU_SRC,
       pr.ID_QUANTUM_SRC,
       pr.STORAGE_AREA_SRC,
       pr.STORAGE_LOCATION_SRC,
       pr.AISLE_SRC,
       pr.ID_LU_TGT,
       pr.STORAGE_AREA_TGT,
       pr.STORAGE_LOCATION_TGT,
       art.txt_label art_txt_label,
       art.unit_stock,
       dte.txt_label dte_txt_label,
       ri.id_goods_receipt,
       ri.num_item,
       pr.id_cre,
       pr.id_upd,
       pr.ts_cre,
       pr.ts_upd
  from bps_process_record pr,
       art_article        art,
       dte_node           dte,
       gr_receiving_item  ri
 where pr.typ_step            = 'SETUP'
   and art.id_client          = pr.id_client
   and art.id_article         = pr.id_article
   and dte.id_context         = pr.id_site
   and dte.id_node            = pr.id_node
   and ri.id_site          (+)= pr.id_ref_1
   and ri.id_client        (+)= pr.id_ref_2
   and ri.id_goods_receipt (+)= pr.id_ref_3
   and ri.num_item         (+)= pr.id_ref_4;
