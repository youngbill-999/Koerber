create or replace view tcs_v_node_sl_asgmt_helper as
select a.id_site
      ,a.storage_area
      ,a.txt_label
      ,a.storage_loc_part1
      ,a.storage_loc_part2
      ,a.storage_loc_part3
      ,a.storage_loc_part4
      ,a.storage_loc_part5
      ,a.storage_loc_part6
      ,min(l.storage_loc_part1) as min_storage_loc_part1
      ,max(l.storage_loc_part1) as max_storage_loc_part1
      ,min(l.storage_loc_part2) as min_storage_loc_part2
      ,max(l.storage_loc_part2) as max_storage_loc_part2
      ,min(l.storage_loc_part3) as min_storage_loc_part3
      ,max(l.storage_loc_part3) as max_storage_loc_part3
      ,min(l.storage_loc_part4) as min_storage_loc_part4
      ,max(l.storage_loc_part4) as max_storage_loc_part4
      ,min(l.storage_loc_part5) as min_storage_loc_part5
      ,max(l.storage_loc_part5) as max_storage_loc_part5
      ,min(l.storage_loc_part6) as min_storage_loc_part6
      ,max(l.storage_loc_part6) as max_storage_loc_part6
  from top_storage_area     a
      ,top_storage_location l
 where a.id_site = l.id_site
   and a.storage_area = l.storage_area
   and a.flg_lu_mgmt = '1'
 group by a.id_site
      ,a.storage_area
      ,a.txt_label
      ,a.storage_loc_part1
      ,a.storage_loc_part2
      ,a.storage_loc_part3
      ,a.storage_loc_part4
      ,a.storage_loc_part5
      ,a.storage_loc_part6;
