create or replace view pac_v_lbl_shipping as
select pa.id_site,

       pa.id_package,
       pa.sscc,
       ad.name_1        as name_1_client,
       ad.name_2        as name_2_client,
       (to_char(ad.street ||' '|| ad.street_num))street_client,
       (to_char(ad.zip ||' '|| ad.city))city_client,
       ad.code_iso_ctry as code_iso_ctry_client,
       adr.name_1,
       adr.name_2,
       (to_char(adr.street ||' '|| adr.street_num))street_receiver,
       (to_char(adr.zip ||' '|| adr.city))city_receiver,
       adr.id_client,
       adc.code_iso_ctry
  from pa_package pa,
       gi_order gi,
       gi_order_address adr,
       ad_address ad,
       ad_country adc
 where
   --pa_package/gi_order
   (pa.typ_ref = 'ORDER'
   or pa.typ_ref = 'ORDER_ITEM')
   and pa.id_ref_1 = gi.id_site (+)
   and pa.id_ref_2 = gi.id_client (+)
   and pa.id_ref_3 = gi.id_order (+)
   and pa.id_ref_4 = gi.num_partial_order (+)
   --gi_order, gi_order_address
   and gi.id_site = adr.id_site (+)
   and gi.id_client = adr.id_client (+)
   and gi.id_order = adr.id_order (+)
   and gi.num_partial_order = adr.num_partial_order (+)
   and adr.typ_addr = 'SHIP'
   --wb_client, ad_address
   and ad.typ_addr = 'GENR'
   and ad.typ_ref = 'CLNT'
   and gi.id_client = ad.id_ref_1 (+)
   --ad_address, ad_country
   and ad.code_iso_ctry = adc.code_iso_ctry (+)
;
