     o.id_client,
     o.id_order,
     o.num_partial_order,
     o.stat,
     o.ts_shipping_target,
     o.wt_target,
     o.vol_target,
     case when o.stat >= '60' then '1' else '0' end as flg_staging_completed,
     c.id_customer,
     c.name,
     c.search_item,
     lsu.id_loading,
     lsu.id_staging_area
from gi_order o
left join pm_customer c
  on c.id_client = o.id_client
 and c.id_customer = o.id_customer
join ship_order_asgmt oa
  on oa.id_site = o.id_site
 and oa.id_client = o.id_client
 and oa.id_order = o.id_order
 and oa.num_partial_order = o.num_partial_order
left join load_loading_staging_usage lsu
  on lsu.id_site = oa.id_site
 and lsu.id_loading = oa.id_loading
order by
     o.id_site,
     o.id_client,
     o.id_order,
     o.num_partial_order
;
