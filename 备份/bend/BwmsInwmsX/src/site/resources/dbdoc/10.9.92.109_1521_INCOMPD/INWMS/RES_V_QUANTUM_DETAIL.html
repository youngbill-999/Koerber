create or replace view res_v_quantum_detail as
select q.id_site,             -- to identify selected quantum
       q.id_quantum,          -- to identify selected quantum
       q.id_client,           -- to identify article
       q.id_article,           -- to identify article
       q.qty_available,       -- to satisfy need
       q.qty_reserved,        -- to check if complete lu
       q.flg_stock_taking as quant_flg_stock_taking,    -- check - missing in default?
       q.id_lu_1,             -- for logging
       q.typ_lu_1,            -- for logging
       q.id_lu_2,             -- for logging
       q.typ_lu_2,            -- for logging
       q.id_lu_3,             -- for logging
       q.typ_lu_3,            -- for logging
       q.storage_area,         -- needed to join storage area
       q.storage_location,     -- needed to join storage location
       --q.storage_area_src,   -- not needed, because no transit
       --q.storage_location_src,    -- not needed because no transit
       q.flg_in_transit,            -- needed for explicit check on transit
       --q.id_transaction,       -- maybe useful for logging
       q.artvar,                 -- needed for default check
       pu.typ_lu as typ_lu_unopened,                -- needed for validation
       pu.qty_per_lu,            --needed for identification of unopened / complete lu
       q.typ_stock,              -- needed for default check
       q.typ_lock,               -- needed for default check
       q.id_batch,               -- needed for default check
       q.typ_special_stock,      -- needed for default check
       q.id_special_stock,       -- needed for default check
       q.stat_quality_control,   -- needed for default check
       q.stat_customs,           -- needed for default check
       q.sep_crit_01,            -- needed for default check
       q.sep_crit_02,            -- needed for default check
       q.sep_crit_03,            -- needed for default check
       q.sep_crit_04,            -- needed for default check
       q.sep_crit_05,            -- needed for default check
       q.sep_crit_06,            -- needed for default check
       q.sep_crit_07,            -- needed for default check
       q.sep_crit_08,            -- needed for default check
       q.sep_crit_09,            -- needed for default check
       q.sep_crit_10,            -- needed for default check
       q.ts_bbd,                 -- needed for sorting
       TRUNC(q.ts_bbd) as dat_bbd,
       -- q.id_goods_receipt,   -- not needed
       -- q.num_goods_receipt_item, -- not needed
       q.ts_goods_receipt,          -- needed for sorting
       TRUNC(q.ts_goods_receipt) as dat_goods_receipt,
       a.stat as art_stat,                       -- needed for default check
       a.stat_erp as art_stat_erp,                   -- needed for default check
       c.flg_sep_crit_01_nullable,
       c.flg_sep_crit_02_nullable,
       c.flg_sep_crit_03_nullable,
       c.flg_sep_crit_04_nullable,
       c.flg_sep_crit_05_nullable,
       c.flg_sep_crit_06_nullable,
       c.flg_sep_crit_07_nullable,
       c.flg_sep_crit_08_nullable,
       c.flg_sep_crit_09_nullable,
       c.flg_sep_crit_10_nullable,
       ar.flg_virtual as area_flg_virtual,               -- needed for default check
       l.flg_block_conv_tec as loc_flg_block_conv_tec,         -- needed for default check
       l.flg_block_store_out as loc_flg_block_store_out,        -- needed for default check
       l.flg_stock_taking as loc_flg_stock_taking,           -- needed for default check
       ai.aisle,
       ai.flg_block_store_out as aisle_flg_block_store_out,       -- needed for default check
       ai.flg_block_disposition as aisle_flg_block_disposition,
       zn.id_store_out_zone,
       zn.id_store_out_group,
       zn.flg_active as asgmt_active,
       oz.flg_active as zone_active,
       ozg.flg_active as group_active
from im_quantum q
     join art_article a on q.id_client = a.id_client and q.id_article = a.id_article
     join wb_client c on q.id_client = c.id_client
     left outer join art_art_packing_unit pu on q.id_site = pu.id_site and q.id_client = pu.id_client and q.id_article = pu.id_article and q.artvar = pu.artvar and '1' = pu.layer
     join top_storage_area ar on q.id_site = ar.id_site and q.storage_area = ar.storage_area
     join top_storage_location l on q.id_site = l.id_site and q.storage_area = l.storage_area and q.storage_location = l.storage_location
     left join top_storage_aisle ai on  l.id_site = ai.id_site and l.aisle = ai.aisle
     left join top_store_out_zn_grp_asgmt zn on  l.id_site = zn.id_site and l.id_store_out_zone = zn.id_store_out_zone
     left join top_store_out_zone oz on zn.id_site = oz.id_site and zn.id_store_out_zone=oz.id_store_out_zone
     left join top_store_out_group ozg on zn.id_site = ozg.id_site and zn.id_store_out_group = ozg.id_store_out_group
--where
--and q.id_client = ?
--and q.id_article = ?
--and zn.id_store_out_group = ?
