create or replace view load_v_load_staging_loc as
(
select l.id_site,
       l.id_loading,
       l.typ_shipment,
       lsu.num_item,
       sa.id_staging_area,
       sa.storage_area,
       sa.storage_location,
       sa.flg_active
  from load_loading l,
       load_loading_staging_usage lsu,
       top_staging_area sa
 where lsu.id_site         = l.id_site
   and lsu.id_loading      = l.id_loading
   and sa.id_site          = lsu.id_site
   and sa.id_staging_area  = lsu.id_staging_area
);
