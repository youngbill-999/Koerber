       pr.PRIORITY,
       pr.TYP_STEP,

       pr.SEP_CRIT_01,

       pr.AISLE_SRC,
       pr.ID_LU_TGT,
       null as id_picking_trip,
       pr.store_out_zone_src,
       null as id_picking_order,
       null as num_sort,
       '0' as flg_clearing,
       pr.id_cre,
       pr.id_upd,
       pr.ts_cre,
       pr.ts_upd
  from bps_process_record pr,
       bps_process_step ps
 where ps.typ_step = 'PICKING'
   and ps.stat_step < '30'
   and pr.id_site = ps.id_site
   and pr.id_record = ps.id_record
