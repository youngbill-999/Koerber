create or replace view pac_v_rep_pack_content_list as
select

   -- HEAD URL    : $URL: svn://ifbsrc1/Bend2.0/IWFComponent/releases/10_wms/wms_packing/snapshot/comp/inst_components/db/dbobj/views/pac_v_rep_pack_content_list.vew $

   -- Changed     : $Id: pac_v_rep_pack_content_list.vew 39435 2019-09-16 09:40:04Z ljaeger $
   -- Changed from: $Author: ljaeger $
   -- Copyrights  : (C) inconso AG, Germany
   -- Description : View to select data for report
   -----------------------------------------------------------------------------
       pa.id_site,
       --package information
       pa.id_package,
       pa.typ_ref,
       pa.id_lu,
       pa.sscc,
       nvl(pa.wt_weighted, nvl(pa.wt_gross, im.wt_gross)) weight,
       bp.unit_wt,
       -- address data
       gi.id_customer,
       adr.typ_addr,
       adr.name_1,
       adr.name_2,
       adr.street,
       adr.street_num,
       adr.city,
       adr.zip,
       adr.code_iso_ctry adr_country,
       cnt.name_local,
       --position data
       art.id_client,
       bp.id_article,
       ads.txt_label,
       bp.ts_bbd,
       bp.id_batch,
       bp.qty,
       art.unit_stock,
       ads.id_locale,
       --site data
       wbadr.code_iso_ctry site_country
from
       pa_package pa,
       pa_package_bps_asgmt pabp,
       bps_process_record bp,
       gi_order_address adr,
       gi_order gi,
       art_article art,
       art_v_art_label ads,
       im_load_unit im,
       ad_address wbadr,
       ad_country cnt
where
   --pa_package restrictions
   (pa.typ_ref = 'ORDER_ITEM' or pa.typ_ref = 'ORDER')
   and pa.stat >= '50' and pa.stat < '80'
   --pa_package/ad_address
   and pa.id_site = wbadr.id_ref_1
   --ad_address restrictions
   and wbadr.typ_ref = 'SITE'
   --pa_package/im_load_unit
   and pa.id_site = im.id_site
   and pa.id_lu = im.id_lu
   --pa_package/pa_package_bps_asgmt
   and pa.id_site = pabp.id_site
   and pa.id_package = pabp.id_package
   --pa_package_bps_asgmt/bps_process_record
   and pabp.id_site = bp.id_site
   and pabp.id_record = bp.id_record
   --bps_process_record restrictions
   and (bp.typ_ref = 'ORDER_ITEM' or bp.typ_ref = 'ORDER')
   and bp.stat < '80'
   --bps_process_record/art_article
   and bp.id_article = art.id_article
   and bp.id_client  = art.id_client
   --art_article/art_v_art_label
   and art.id_client = ads.id_client
   and art.id_article = ads.id_article
   --bps_process_record/gi_order
   and bp.id_ref_1 = gi.id_site (+)
   and bp.id_ref_2 = gi.id_client (+)
   and bp.id_ref_3 = gi.id_order (+)
   and bp.id_ref_4 = gi.num_partial_order (+)
   --gi_order/gi_order_address
   and gi.id_site = adr.id_site (+)
   and gi.id_client = adr.id_client (+)
   and gi.id_order = adr.id_order (+)
   and gi.num_partial_order = adr.num_partial_order (+)
   --gi_order_address restrictions
   and adr.typ_addr (+) = 'SHIP'
   --gi_order_address/ad_country
   and adr.code_iso_ctry = cnt.code_iso_ctry (+)
;
