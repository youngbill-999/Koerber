create or replace view pa_v_package_bps_asgmt as
select pr.id_site
      ,asgmt.id_package
      ,pr.id_record
      ,pr.id_client
      ,pr.id_article
      ,pr.id_batch
      ,pr.ts_bbd
      ,pr.qty
      ,a.txt_label
      ,a.unit_stock
      ,asgmt.id_package_item
      ,asgmt.ts_pack
      ,asgmt.id_user_pack
      ,asgmt.id_terminal_pack
      ,asgmt.id_workcenter_pack
from bps_process_record pr
join pa_package_bps_asgmt asgmt on asgmt.id_site = pr.id_site and asgmt.id_record = pr.id_record
join art_article a on a.id_client = pr.id_client and a.id_article = pr.id_article
group by
      pr.id_site
      ,asgmt.id_package
      ,pr.id_record
      ,pr.id_client
      ,pr.id_article
      ,pr.id_batch
      ,pr.ts_bbd
      ,pr.qty
      ,a.txt_label
      ,a.unit_stock
      ,asgmt.id_package_item
      ,asgmt.ts_pack
      ,asgmt.id_user_pack
      ,asgmt.id_terminal_pack
      ,asgmt.id_workcenter_pack;
