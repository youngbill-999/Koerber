create or replace view stm_v_stock_taking_item_rec as
select
       i.id_site,
       i.id_stock_taking_list,
       i.num_item,
       i.stat,
       i.typ_item,
       i.storage_area,
       i.storage_location,
       r.id_client,
       r.id_article,
       a.txt_label,
       a.unit_stock,
       r.qty,
       c.qty_actual,
       c_last.qty_actual as qty_last_count,
       (c.qty_actual - r.qty) as qty_difference,
       r.id_record,
       r.id_lu_src as id_lu,
       l.typ_lu,
       r.id_quantum_src as id_quantum,
       r.id_batch,
       r.ts_bbd,
       c.id_user_count,
       c.ts_count,
       i.id_cre,
       i.ts_cre,
       i.id_upd,
       i.ts_upd
from stm_stock_taking_item i
join bps_process_record r on
     r.typ_ref = 'STOCK_TAKING_ITEM'
 and r.id_ref_1 = i.id_site
 and r.id_ref_2 = i.id_stock_taking_list
 and r.id_ref_3 = i.num_item
 and r.id_ref_4 is null
 and r.id_ref_5 is null
 and r.id_ref_6 is null
left outer join Art_Article a on
      r.id_client = a.id_client
  and r.id_article = a.id_article
left outer join Im_load_unit l on
      r.id_site = l.id_site and
      r.id_lu_src = l.id_lu
left outer join (select * from stm_stock_taking_count c where c.cnt_count =
     (select max(sub.cnt_count) from stm_stock_taking_count sub
     where sub.id_site= c.id_site and
           sub.id_record = c.id_record)) c
     on c.id_site = r.id_site and
        c.id_record = r.id_record
left outer join stm_stock_taking_count c_last on c_last.id_site = r.id_site and c_last.id_record = r.id_record and c_last.cnt_count = c.cnt_count - 1;
