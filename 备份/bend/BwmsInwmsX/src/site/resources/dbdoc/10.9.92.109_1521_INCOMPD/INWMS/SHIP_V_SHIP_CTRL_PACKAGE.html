create or replace view ship_v_ship_ctrl_package as
select
     p.id_site,

     p.id_package,
     p.stat as package_stat,
     p.id_lu,
     p.typ_ref,
     p.id_ref_1,
     p.id_ref_2,
     p.id_ref_3,
     p.id_ref_4,
     o.id_client,
     o.id_order,
     o.num_partial_order,
     o.stat as order_stat,
     o.ts_shipping_target,
     c.id_customer,
     c.name,
     lu.typ_lu,
     lu.sscc,
     lu.storage_area_src,
     lu.storage_location_src,
     lu.storage_area,
     lu.storage_location,
     lu.flg_in_transit,
     coalesce(p.wt_weighted, p.wt_gross, lu.wt_gross) as wt,
     coalesce(p.vol, lut.vol_inner) as vol,
     lsu.id_staging_area,
     lsu.id_loading
from pa_package p
join im_load_unit lu on lu.id_site = p.id_site and lu.id_lu = p.id_lu
join top_lu_type lut on lut.id_site = lu.id_site and lut.typ_lu = lu.typ_lu
join ship_order_asgmt oa
  on oa.id_site = p.id_ref_1
 and oa.id_client = p.id_ref_2
 and oa.id_order = p.id_ref_3
 and oa.num_partial_order = p.id_ref_4
join gi_order o
  on o.id_site = oa.id_site
 and o.id_client = oa.id_client
 and o.id_order = oa.id_order
 and o.num_partial_order = oa.num_partial_order
left join pm_customer c
  on c.id_client = o.id_client
 and c.id_customer = o.id_customer
left join load_loading_staging_usage lsu
  on lsu.id_site = oa.id_site
 and lsu.id_loading = oa.id_loading
order by
     p.id_site,
     p.id_package
;
