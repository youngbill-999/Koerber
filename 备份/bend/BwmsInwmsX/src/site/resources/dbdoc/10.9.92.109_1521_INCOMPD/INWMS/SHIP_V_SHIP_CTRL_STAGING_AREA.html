create or replace view ship_v_ship_ctrl_staging_area as
select
     s.id_site,

     s.id_staging_area,
     s.id_site || '@' || s.id_staging_area || '@' ||  l.id_loading as id_technical,
     sl.pct_occupied,
     sl.stat_occupied,
     l.id_loading,
     l.typ_shipment,
     l.ts_cut_off,
     l.stat,
     c.id_carrier,
     c.name,
     c.search_item,
     tv.id_transport_vehicle,
     tv.name_driver,
     tv.truck_license_plate,
     coalesce(cas.count_assigned_orders, 0) as count_assigned_orders,
     coalesce(cas.count_staged_orders, 0) as count_staged_orders,
     coalesce(cas.count_loaded_orders, 0) as count_loaded_orders,
     coalesce(cas.count_assigned_packages, 0) as count_assigned_packages,
     coalesce(cas.count_staged_packages, 0) as count_staged_packages,
     coalesce(cas.count_loaded_packages, 0) as count_loaded_packages,
     coalesce(cas.sum_wt_assigned, 0) as sum_wt_assigned,
     coalesce(cas.sum_wt_staged, 0) as sum_wt_staged,
     coalesce(cas.sum_wt_loaded, 0) as sum_wt_loaded,
     coalesce(cas.sum_vol_assigned, 0) as sum_vol_assigned,
     coalesce(cas.sum_vol_staged, 0) as sum_vol_staged,
     coalesce(cas.sum_vol_loaded, 0) as sum_vol_loaded,
     st.cl_shipment
from top_staging_area s
join top_storage_location sl on sl.id_site = s.id_site and sl.storage_area = s.storage_area and sl.storage_location = s.storage_location
left join ship_v_ship_ctrl_count_and_sum cas on cas.id_site = s.id_site and cas.id_staging_area = s.id_staging_area
left join load_loading l on l.id_site = cas.id_site and l.id_loading = cas.id_loading
left join load_shipment_typ st on st.id_site = l.id_site and st.typ_shipment = l.typ_shipment
left join pm_carrier c on c.id_site = st.id_site and c.id_carrier = st.id_carrier
left join tm_transport_vehicle tv on tv.id_site = l.id_site and tv.id_transport_vehicle = l.id_transport_vehicle
where s.flg_active = '1'
  and (l.stat is null or l.stat != '90')
group by
     s.id_site,
     s.id_staging_area,
     sl.pct_occupied,
     sl.stat_occupied,
     l.id_loading,
     l.typ_shipment,
     l.ts_cut_off,
     l.stat,
     c.id_carrier,
     c.name,
     c.search_item,
     tv.id_transport_vehicle,
     tv.name_driver,
     tv.truck_license_plate,
     cas.count_assigned_orders,
     cas.count_staged_orders,
     cas.count_loaded_orders,
     cas.count_assigned_packages,
     cas.count_staged_packages,
     cas.count_loaded_packages,
     cas.sum_wt_assigned,
     cas.sum_wt_staged,
     cas.sum_wt_loaded,
     cas.sum_vol_assigned,
     cas.sum_vol_staged,
     cas.sum_vol_loaded,
     st.cl_shipment
order by
     s.id_site,
     s.id_staging_area,
     l.id_loading
;
